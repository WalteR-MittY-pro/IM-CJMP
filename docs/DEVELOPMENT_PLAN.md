#### 7.11.2 登录页面组件结构

```cangjie
@Component
class LoginPage {
    @Prop var onLoginSuccess: () -> Unit

    @State
    var phone: String = ""

    @State
    var code: String = ""

    @State
    var errorMessage: String = ""

    func build() {
        Column() {
            // App Logo
            Image($r("media.icon"))
                .width(96)
                .height(96)

            // 手机号输入框
            TextInput({ placeholder: "请输入手机号", text: this.phone })
                // ... 样式

            // 验证码输入框
            TextInput({ placeholder: "请输入验证码", text: this.code })
                // ... 样式

            // 登录按钮
            Button("登录")
                .onClick({ => validateAndLogin() })

            // 错误提示
            if (this.errorMessage != "") {
                Text(this.errorMessage)
                    .foregroundColor(Color.Red)
            }
        }
    }
}
```
